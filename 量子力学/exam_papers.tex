    \expval{x^2} = & \frac{2}{m \omega^2} \expval{V(x)} = \frac{2}{m \omega^2} \frac{1}{2} (n + \frac{1}{2}) \hbar \omega = \frac{3}{2} \frac{\hbar}{m \omega}\\
    \expval{p^2} = & 2 m \expval{T} = 2 m \frac{1}{2} (n + \frac{1}{2}) \hbar \omega = \frac{3}{2} m \hbar \omega
